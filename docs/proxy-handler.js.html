proxy-handler.js

/**
 * @module handler
 * @requires config
 * @requires helpers
 * @requires request-promise-factory
 */

const requestPromiseFactory = require('./request-promise-factory');
const helpers = require('./helpers');
const getMd5Hash = helpers.getMd5Hash;
const sortAndConcatValues = helpers.sortAndConcatValues;
const config = require('./config');

const handler = {};

handler.get = function(target, methodName) {
  if (!target[methodName] && typeof target[methodName] !== 'function') {
    throw new Error(`${methodName} method doesn't exist in API`);
  }
  return new Proxy(target[methodName], {
    apply: function(targetMethod, thisArg, args) {
      const parsedArgs = targetMethod(...args);
      for (let i in parsedArgs) {
        if (parsedArgs.hasOwnProperty(i)) {
          if (parsedArgs[i] === undefined) {
            throw new Error('missing required parameter');
          }
        }
      }
      if (config.testMode) {
        parsedArgs.test = 1;
      }
      parsedArgs.key = config.keys.publicKey;
      const tempArgs = {};
      Object.assign(tempArgs, parsedArgs);
      tempArgs.version = config.apiVersion;
      tempArgs.action = methodName;
      parsedArgs.sum = getMd5Hash(sortAndConcatValues(tempArgs) +
        config.keys.privateKey);
      return requestPromiseFactory(methodName, parsedArgs);
    },
  });
};

module.exports = handler;
