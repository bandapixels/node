/**
 * !!! Default methods of this interface used to validate provided parameters !!!
 *
 * @see {@link https://www.epochta.ru/products/sms/v3.php}
 *
 * @interface APIMethods
 *
 * @description Inrerface for methods provided by ePochta SMS service API. </br>
 * All arguments are positional! To provide not first optional argument, omitting previous optional arguments use <code>null</code> as stub.
 * @example <caption> Use <code>null</code> as stub </caption>
 * // Ways to get first ten address books:
 * correct: client.getAddressbook(null, null, 10).then...
 * wrong: client.getAddressbook(10).then...
 */
const APIMethods = {
  /**
   * Creates new address book with provided name and description[optional].
   * @instance
   * @param {string} name - Name of the address book
   * @param {?string} [description] - Description of the address book
   * @return {Promise<number|Error>} - Unique identifier of the created address book or throws Error
   */
  addAddressbook(name, description=null) {
    return {
      args: {name, description},
      returns: ['addressbook_id'],
      };
  },

  /**
   * Deletes an address book by id.
   * @instance
   * @param {number} id - id of the address book to delete
   * @return {Promise<boolean|Error>} - In case of success returns true, otherwise throws Error
   */
  delAddressbook(id) {
    return {
      args: {id},
      returns: ['successful'],
    };
  },

  /**
   * Returns address book data.
   * @instance
   * @param {?number} [idAddressBook] - id of the address book to get data. If not provided, returns all address books data
   * @param {?number} [from] - return address books starting at 'from'
   * @param {?number} [offset] - number of address books to return
   * @return {Promise<APIResponseObjects.AddressBook|Array<APIResponseObjects.AddressBook>|Error>} address book/books data
   */
  getAddressbook(idAddressBook=null, from=null, offset=null) {
    return {idAddressBook};
  },
};

module.exports = APIMethods;
