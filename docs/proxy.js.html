/**
 * @module
 */
const request = require('request');
const config = require('./config');
const crypto = require('crypto');
const errors = require('./errors');

const clientHandler = {};

clientHandler.get = function getTrap(target, methodName) {
  if (!target[methodName] && typeof target[methodName] !== 'function') {
    throw new Error(`${methodName} method doesn't exist in API`);
  }
  return new Proxy(target[methodName], clientMethodsHandler);
};

const clientMethodsHandler = {};

clientMethodsHandler.apply = function applyTrap(targetMethod, thisArg, args) {
  const parsedArgs = targetMethod(...args);
  for (let i in parsedArgs) {
    if (parsedArgs.hasOwnProperty(i)) {
      if (parsedArgs[i] === undefined) {
        throw new Error('missing required parameter');
      }
    }
  }
  if (config.testMode) {
    parsedArgs.test = 1;
  }
  parsedArgs.key = config.keys.publicKey;
  const tempArgs = {};
  Object.assign(tempArgs, parsedArgs);
  tempArgs.version = config.apiVersion;
  tempArgs.action = targetMethod.name;
  parsedArgs.sum = getMd5Hash(concatArgs(tempArgs) +
    config.keys.privateKey);
  return new Promise(function(resolve, reject) {
    request({
      method: config.request.method,
      baseUrl: config.request.baseUrl,
      uri: targetMethod.name,
      body: parsedArgs,
      json: true,
    }, function(error, response, body) {
      if (error) {
        return reject(error);
      }
      if (response && response.statusCode !== config.responseStatusCode) {
        return reject(new errors.ResponseCodeError(response.statusCode));
      }
      if (body.error) {
        return reject(body.error);
      }
      resolve(body.result);
    });
  });
};

/**
 * Calculates MD5 hash of provided data
 * @param {string} stringData
 * @return {string}
 */
function getMd5Hash(stringData) {
  return crypto.createHash('md5').update(stringData).digest('hex');
}

/**
 * Concatenation of request arguments according to ePochta API v3.0 algorithm
 * @see {@link https://www.epochta.ru/products/sms/v3.php}
 * @param {Object} obj - API request arguments
 * @return {string} Checksum
 */
function concatArgs(obj) {
  return Object.keys(obj).sort()
    .reduce((sum, key) => sum + (Array.isArray(obj[key]) ?
      'Array' : obj[key]), '');
}

module.exports = clientHandler;
